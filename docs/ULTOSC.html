ULTOSC


DESCRIPTION: ULTOSC
SYNTAX: COMMAND = ULTOSC
STEP = <VARIABLE NAME>
HIGH = <VARIABLE NAME>
LOW = <VARIABLE NAME>
CLOSE = <VARIABLE NAME>
NAME = <VARIABLE NAME>
PERIOD_SHORT = <INTEGER>
PERIOD_MED = <INTEGER>
PERIOD_LONG = <INTEGER>
RETURNS: 'OK' = Success
'ERROR' = Failure
EXAMPLE: COMMAND=ULTOSC;
STEP=ultosc;
HIGH=high;
LOW=low;
CLOSE=close;
NAME=ultosc;
PERIOD_SHORT=7;
PERIOD_MED=14;
PERIOD_LONG=28
