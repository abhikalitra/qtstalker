ROC


DESCRIPTION: Rate Of Change
SYNTAX: PLUGIN = ROC
NAME = <VARIABLE NAME>
METHOD = <METHOD>
  1. ROC
  2. ROCP
  3. ROCR
  4. ROCR100
INPUT = <VARIABLE NAME>
PERIOD = <INTEGER>
RETURNS: '0' = Success
'ERROR' = Failure
EXAMPLE: PLUGIN=ROC,METHOD=ROC,NAME=roc,INPUT=close,PERIOD=10
