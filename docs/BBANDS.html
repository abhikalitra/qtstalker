BBANDS


DESCRIPTION: Bollinger Bands
SYNTAX: COMMAND = BBANDS
STEP = <VARIABLE NAME>
INPUT = <VARIABLE NAME>
NAME_UPPER = <VARIABLE NAME>
NAME_MIDDLE = <VARIABLE NAME>
NAME_LOWER = <VARIABLE NAME>
PERIOD = <INTEGER>
MA_TYPE = <MA TYPE>
DEV_UP = <DECIMAL>
DEV_DOWN = <DECIMAL>
RETURNS: 'OK' = Success
'ERROR' = Failure
EXAMPLE: COMMAND=BBANDS; STEP=bbands; INPUT=close; NAME_UPPER=bbu; NAME_MIDDLE=bbm; NAME_LOWER=bbl; PERIOD=20; MA_TYPE=EMA; DEV_UP=2; DEV_DOWN=2
