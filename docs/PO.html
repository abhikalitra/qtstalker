PO


DESCRIPTION: Price Oscillator
SYNTAX: PLUGIN = PO
NAME = <VARIABLE NAME>
METHOD =
  1. APO
  2. PPO
INPUT = <VARIABLE NAME>
PERIOD_FAST = <INTEGER>
PERIOD_SLOW = <INTEGER>
MA_TYPE = <MA TYPE>
RETURNS: '0' = Success
'ERROR' = Failure
EXAMPLE: PLUGIN=PO,NAME=po,METHOD=APO,INPUT=close,PERIOD_FAST=12,PERIOD_SLOW=26,MA_TYPE=EMA
