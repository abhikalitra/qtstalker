STOCH_FAST


DESCRIPTION: Fast Stochastic Oscillator
SYNTAX: COMMAND = STOCH_FAST
STEP = <VARIABLE NAME>
HIGH = <VARIABLE NAME>
LOW = <VARIABLE NAME>
CLOSE = <VARIABLE NAME>
NAME_FASTK = <VARIABLE NAME>
NAME_FASTD = <VARIABLE NAME>
PERIOD_FASTK = <INTEGER>
PERIOD_FASTD = <INTEGER>
MA_TYPE = <MA TYPE>
RETURNS: 'OK' = Success
'ERROR' = Failure
EXAMPLE: COMMAND=STOCH_FAST;
STEP=stochf;
HIGH=high;
LOW=low;
CLOSE=close;
NAME_FASTK=%K;
NAME_FASTD=%D;
PERIOD_FASTK=3;
PERIOD_FASTD=3;
MA_TYPE=EMA
