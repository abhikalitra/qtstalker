STOCH_FAST


DESCRIPTION: Fast Stochastic Oscillator
SYNTAX: PLUGIN = STOCH_FAST
INPUT_HIGH = <VARIABLE NAME>
INPUT_LOW = <VARIABLE NAME>
INPUT_CLOSE = <VARIABLE NAME>
NAME_FASTK = <VARIABLE NAME>
NAME_FASTD = <VARIABLE NAME>
PERIOD_FASTK = <INTEGER>
PERIOD_FASTD = <INTEGER>
MA_TYPE_FASTD = <MA TYPE>
RETURNS: '0' = Success
'ERROR' = Failure
EXAMPLE: PLUGIN=STOCH_FAST, INPUT_HIGH=high, INPUT_LOW=low, INPUT_CLOSE=close, NAME_FASTK=%K, NAME_FASTD=%D, PERIOD_FASTK=3, PERIOD_FASTD=3, MA_TYPE_FASTD=EMA
