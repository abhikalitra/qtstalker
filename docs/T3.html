T3


DESCRIPTION: Triple Exponential Moving Average
SYNTAX: COMMAND = T3
STEP = <VARIABLE NAME>
NAME = <VARIABLE NAME>
INPUT = <VARIABLE NAME>
PERIOD = <INTEGER>
VFACTOR = <DECIMAL>
RETURNS: 'OK' = Success
'ERROR' = Failure
EXAMPLE: COMMAND=T3;
STEP=t3;
NAME=t3;
INPUT=close;
PERIOD=5;
VFACTOR=0.7
