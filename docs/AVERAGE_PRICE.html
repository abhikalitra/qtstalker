AVERAGE_PRICE


DESCRIPTION: Average Price calculation is ((Open + High + Low + Close) / 4)
SYNTAX: PLUGIN = AVERAGE_PRICE
INPUT_OPEN = <VARIABLE NAME>
INPUT_HIGH = <VARIABLE NAME>
INPUT_LOW = <VARIABLE NAME>
INPUT_CLOSE = <VARIABLE NAME>
NAME = <VARIABLE NAME>
RETURNS: '0' = Success
'ERROR' = Failure
EXAMPLE: PLUGIN=AVERAGE_PRICE, INPUT_OPEN=open, INPUT_HIGH=high, INPUT_LOW=low, INPUT_CLOSE=close, NAME=ap
