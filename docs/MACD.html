MACD


DESCRIPTION: Moving Average Convergence Divergence
SYNTAX: COMMAND = MACD
STEP = <VARIABLE NAME>
INPUT = <VARIABLE NAME>
NAME_MACD = <VARIABLE NAME>
NAME_SIGNAL = <VARIABLE NAME>
NAME_HIST = <VARIABLE NAME>
PERIOD_FAST = <INTEGER>
MA_TYPE_FAST = <MA TYPE>
PERIOD_SLOW = <INTEGER>
MA_TYPE_SLOW = <MA TYPE>
PERIOD_SIGNAL = <INTEGER>
MA_TYPE_SIGNAL = <MA TYPE>
RETURNS: 'OK' = Success
'ERROR' = Failure
EXAMPLE: COMMAND=MACD;
STEP=macd;
INPUT=close;
NAME_MACD=macd;
NAME_SIGNAL=sig;
NAME_HIST=hist;
PERIOD_FAST=12;
MA_TYPE_FAST=EMA;
PERIOD_SLOW=26;
MA_TYPE_SLOW=EMA;
PERIOD_SIGNAL=9;
MA_TYPE_SIGNAL=EMA
