CHART_OBJECT_HLINE


DESCRIPTION: Creates a horizontal line chart object.

SYNTAX: PLUGIN = CHART_OBJECT_HLINE
TYPE = <TYPE>
  • RO - read only, not user interactive, not saved to disk
  • RW - read write, user interactive, saved to disk
NAME = <VARIABLE NAME>
INDICATOR = <INDICATOR NAME>
EXCHANGE = <EXCHANGE>
SYMBOL = <SYMBOL>
PRICE = <DECIMAL>
COLOR = <COLOR>
RETURNS: '0' = Success
'ERROR' = Failure
EXAMPLE: PLUGIN=CHART_OBJECT_HLINE,TYPE=RW,NAME=hline1,INDICATOR=Bars,EXCHANGE=XNYS,SYMBOL=GOOG,PRICE=450.5,COLOR=white
