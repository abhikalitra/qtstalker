QUOTE_DATABASE_GET


DESCRIPTION: Load quotes from the database.
SYNTAX: PLUGIN = QUOTE_DATABASE_GET
NAME_OPEN = <VARIABLE NAME> - Optional use
NAME_HIGH = <VARIABLE NAME> - Optional use
NAME_LOW = <VARIABLE NAME> - Optional use
NAME_CLOSE = <VARIABLE NAME> - Optional use
NAME_VOLUME = <VARIABLE NAME> - Optional use
NAME_OI = <VARIABLE NAME> - Optional use
EXCHANGE = <EXCHANGE>
SYMBOL = <SYMBOL>
LENGTH = <BAR LENGTH>
  1. 1 - Minute
  2. 5 - Minute
  3. 10 - Minute
  4. 15 - Minute
  5. 30 - Minute
  6. 60 - Minute
  7. D - Daily
  8. W - Weekly
  9. M - Monthly
RANGE = <DATE RANGE>
  1. 1 Day
  2. 1 Week
  3. 1 Month
  4. 3 Months
  5. 6 Months
  6. 1 Year
  7. 2 Years
  8. 5 Years
  9. 10 Years
  10. 25 Years
  11. 50 Years
  12. All
RETURNS: '0' = Success
'ERROR' = Failure
EXAMPLE: PLUGIN=QUOTE_DATABASE_GET,NAME_CLOSE=close,NAME_VOLUME=volume,EXCHANGE=XNYS,SYMBOL=GOOG,LENGTH=D,RANGE=1 Year
