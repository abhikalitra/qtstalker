STOCH_SLOW


DESCRIPTION: Slow Stochastic Oscillator
SYNTAX: COMMAND = STOCH_SLOW
STEP = <VARIABLE NAME>
HIGH = <VARIABLE NAME>
LOW = <VARIABLE NAME>
CLOSE = <VARIABLE NAME>
NAME_SLOWK = <VARIABLE NAME>
NAME_SLOWD = <VARIABLE NAME>
PERIOD_FASTK = <INTEGER>
PERIOD_SLOWK = <INTEGER>
MA_TYPE_SLOWK = <MA TYPE>
PERIOD_SLOWD = <INTEGER>
MA_TYPE_SLOWD = <MA TYPE>
RETURNS: 'OK' = Success
'ERROR' = Failure
EXAMPLE: COMMAND=STOCH_SLOW;
STEP=stochs;
HIGH=high;
LOW=low;
CLOSE=close;
NAME_SLOWK=%K;
NAME_SLOWD=%D;
PERIOD_FASTK=5;
PERIOD_SLOWK=3;
MA_TYPE_SLOWK=EMA;
PERIOD_SLOWD=3;
MA_TYPE_SLOWD=EMA
