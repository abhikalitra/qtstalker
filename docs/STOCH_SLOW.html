STOCH_SLOW


DESCRIPTION: Slow Stochastic Oscillator
SYNTAX: PLUGIN = STOCH_SLOW
NAME_SLOWK = <VARIABLE NAME>
NAME_SLOWD = <VARIABLE NAME>
PERIOD_FASTK = <INTEGER>
PERIOD_SLOWK = <INTEGER>
MA_TYPE_SLOWK = <MA TYPE>
PERIOD_SLOWD = <INTEGER>
MA_TYPE_SLOWD = <MA TYPE>
RETURNS: '0' = Success
'ERROR' = Failure
EXAMPLE: PLUGIN=STOCH_SLOW,NAME_SLOWK=%K,NAME_SLOWD=%D,PERIOD_FASTK=3,PERIOD_SLOWK=3,MA_TYPE_SLOWK=EMA,PERIOD_SLOWD=3,MA_TYPE_SLOWD=EMA
